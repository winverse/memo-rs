# run
```bash
# add 
cargo run add <name>
cargo run add <name> -e <email> # if needs the email infomation

# list
cargo run list

# remove
cargo run remove <id>

# update
cargo run edit <id> <name>
cargo run edit <id> <name> -e <email> # if needs the email infomation

# search
cargo run search <name>
```
